        private function showElem(elem:*, show:Boolean):void {
            elem.visible = show;
            elem.includeInLayout = show;
        }
        
        public function set infoDP(dp:ArrayCollection):void {
            _infodp = dp;

            init();

            if(dp.length > 0) {
                var valuekeys:Array = new Array();
                var displayNames:Array = new Array();
                var obj:Object = dp[0];
                for(var key:String in obj) {
                    if(key == "field") {
                        
                    } else if(key.substr(0, 5) == "label") {
                        if(key.length == 5) {
                            displayNames[0] = obj[key];
                        } else {
                            displayNames[parseInt(key.substring(5))] = obj[key];
                        }
                    } else {
                        valuekeys.push(key);
                    } 
                }
                //var displayName:String = obj["label"];
                initChartSeries(valuekeys, displayNames);
            } else {
                Alert.show("No information provided for charts.");
            }
        }
        
        private function initChartSeries(valuekeys:Array, displayNames:Array):void {
            var colchartseries:Array = new Array();
            var barchartseries:Array = new Array();
            var linechartseries:Array = new Array();
            var piechartseries:Array = new Array();
            
            var i:int = 0;

            for each(var valuekey:String in valuekeys) {
                var displayName:String = displayNames[i];

                // Column Chart
                var colseries:ColumnSeries = new ColumnSeries();
                colseries.xField = "field";
                colseries.yField = valuekey;
                if(displayName != null) {
                    colseries.displayName = displayName;
                }
                colseries.setStyle("labelPosition", "inside");
                colchartseries.push(colseries);
                
                // Bar Chart
                var barseries:BarSeries = new BarSeries();
                barseries.xField = valuekey;
                barseries.yField = "field";
                if(displayName != null) {
                    barseries.displayName = displayName;
                }
                barseries.setStyle("labelPosition", "inside");
                barchartseries.push(barseries);
                
                // Line Chart
                var lineseries:LineSeries = new LineSeries();
                lineseries.xField = "field";
                lineseries.yField = valuekey;
                if(displayName != null) {
                    lineseries.displayName = displayName;
                }
                linechartseries.push(lineseries);
                
                // Pie Chart
                var pieseries:PieSeries = new PieSeries();
                pieseries.field = valuekey;
                pieseries.nameField = "field";
                pieseries.setStyle("labelPosition", "callout");
                pieseries.explodeRadius = .12;
                piechartseries.push(pieseries);

                if(displayNames.length > 1) {
                    i++;
                }
            }
            
            colchart.series = colchartseries;
            barchart.series = barchartseries;
            linechart.series = linechartseries;
            piechart.series = piechartseries;
        }

// ActionScript file
import components.*;

import mx.controls.Alert;
import mx.events.IndexChangedEvent;

[Bindable]
public var layer_tablename:String = '';//Application.application.parameters.layer_tablename;

private var cmpFormMetaLayer:FormMetaLayer;
private var cmpFormLayerColumnDesc:FormLayerColumnDesc;
private var cmpFormThemeMapping:FormThemeMapping;
private var cmpFormLinkTables:FormLinkTables;
private var cmpFormGlobalResourceMapping:FormGlobalResourceMapping;

private function init():void
{
    if(Application.application.parameters.layer_tablename)
    {
        layer_tablename = Application.application.parameters.layer_tablename;
    }
    else
    {
        //layer_tablename = "lyr_39_bangalore_road_plots";
        Alert.show("Required parameters are not set.", "Error");
        tabNavi.visible = false;
        return;
    }
    
    cmpFormMetaLayer = new FormMetaLayer();
    cmpFormMetaLayer.layer_tablename = layer_tablename;
    tabMetaLayer.addChild(cmpFormMetaLayer);

    cmpFormLayerColumnDesc = new FormLayerColumnDesc();
    cmpFormLayerColumnDesc.layer_tablename = layer_tablename;
    tabLayerColDesc.addChild(cmpFormLayerColumnDesc);

    cmpFormThemeMapping = new FormThemeMapping();
    cmpFormThemeMapping.layer_tablename = layer_tablename;
    tabThemeMapping.addChild(cmpFormThemeMapping);

    cmpFormLinkTables = new FormLinkTables();
    cmpFormLinkTables.layer_tablename = layer_tablename;
    tabLinkTables.addChild(cmpFormLinkTables);
    
    cmpFormGlobalResourceMapping = new FormGlobalResourceMapping();
    cmpFormGlobalResourceMapping.tablename = layer_tablename;
    cmpFormGlobalResourceMapping.table_type = 'LAYER';
    tabLayerGlobalResourceMapping.addChild(cmpFormGlobalResourceMapping);
        
    cmpFormMetaLayer.init();
}

private function tabChanged(e:IndexChangedEvent):void
{
    switch(tabNavi.selectedChild.id)
    {
        case 'tabMetaLayer':
            if(!cmpFormMetaLayer.dataFetched)
                cmpFormMetaLayer.init();
        break;
        case 'tabLayerColDesc':
            if(!cmpFormLayerColumnDesc.dataFetched)
                cmpFormLayerColumnDesc.init();
        break;
        case 'tabThemeMapping':
            if(!cmpFormThemeMapping.dataFetched)
                cmpFormThemeMapping.init();
        break;
        case 'tabLinkTables':
            if(!cmpFormLinkTables.dataFetched)
                cmpFormLinkTables.init();
        case 'tabLayerGlobalResourceMapping':
            if(!cmpFormGlobalResourceMapping.dataFetched)
                cmpFormGlobalResourceMapping.init();
        break;
    }
}

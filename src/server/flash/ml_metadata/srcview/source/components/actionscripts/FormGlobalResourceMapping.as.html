import flash.events.ContextMenuEvent;
import flash.ui.ContextMenu;

import mx.controls.Alert;
import mx.events.CloseEvent;
import mx.managers.PopUpManager;
import mx.rpc.events.ResultEvent;
import mx.utils.ObjectProxy;

[Bindable]
public var tablename:String;

[Bindable]
public var table_type:String; 

[Bindable]
public var dataFetched:Boolean = false;

[Bindable]
private var resourceMapping:Array;

[Bindable]
private var cm:ContextMenu;

public function init():void {
    cm = new ContextMenu();
    cm.hideBuiltInItems();
    var cmEdit:ContextMenuItem = new ContextMenuItem("Edit mapping");
    cm.customItems.push(cmEdit);
    cmEdit.addEventListener(ContextMenuEvent.MENU_ITEM_SELECT, onEditMapping);
    var cmDelete:ContextMenuItem = new ContextMenuItem("Delete mapping");
    cm.customItems.push(cmDelete);
    cmDelete.addEventListener(ContextMenuEvent.MENU_ITEM_SELECT, onDeleteMapping);
    //dg.contextMenu = cm;
    
    getGlobalResourceMapping();
}

private function onEditMapping(e:ContextMenuEvent):void {
    if (dgGlobalResourceMapping.selectedIndex >= 0) {
        var obj:Object = dgGlobalResourceMapping.selectedItem;
        var mapping:Object = new Object();
        mapping.id = obj.id;
        mapping.table_column = removeSingleQuotes(obj.table_column);
        mapping.resource_tablename = obj.resource_tablename;
        mapping.resource_column = removeSingleQuotes(obj.resource_column);
        editResourceMapping(mapping);
    } else {
        Alert.show("Select an item to edit","ATTENTION");
    }
}

private function onDeleteMapping(e:ContextMenuEvent):void {
    if (dgGlobalResourceMapping.selectedIndex >= 0) {
        Alert.yesLabel = "Yes";
        Alert.noLabel = "No";
        Alert.show("Are you sure you want to delete the mapping?", "Confirmation: Delete Mapping", 3, this, deleteConfirmationHandler); 
    } else {
        Alert.show("Select an item to delete","ATTENTION");
    }
}

// Event handler function for displaying the selected Alert button.
private function deleteConfirmationHandler(event:CloseEvent):void {
    if (event.detail==Alert.YES) {
        BlockUI();
        var url:String = srvr_url + ml_data_file;
        var method:String = "POST";
        var obj:Object = dgGlobalResourceMapping.selectedItem;
        var mapping_id:int = obj.id;
        var rtablename:String = obj.resource_tablename;
        var tcolumn:String = obj.table_column;
        var rcolumn:String = obj.resource_column;
        var xmlParams:XML = 
            <request>
                <action>deleteGlobalResourceMapping</action>
                <params>
                    <table tablename={tablename} type={table_type}>
                        <mapping id={mapping_id} resource_tablename={rtablename} resource_column={rcolumn} table_column={tcolumn}/>
                    </table>
                </params>
            </request>;
        var params:Object = new Object();
        params.request = xmlParams;
        useHttpService(url, method, params, null, srvcDeleteGlobalResourceMappingResultHandler, srvcHTTPFaultHandler);
    }
}
            
private function srvcDeleteGlobalResourceMappingResultHandler(e:ResultEvent):void {
    if(!chkError(e)) {
        Alert.show(e.result.response.error.msg, "Response");
        getGlobalResourceMapping();
    }
}

public function getGlobalResourceMapping():void {
    BlockUI();
    var url:String = srvr_url + ml_data_file;
    var method:String = "POST";
    var xmlParams:XML = 
        <request>
            <action>getGlobalResourceMapping</action>
            <params>
                <table type={table_type}>{tablename}</table>
            </params>
        </request>;
    var params:Object = new Object();
    params.request = xmlParams;
    useHttpService(url, method, params, null, srvcGetGlobalResourceMappingResultHandler, srvcHTTPFaultHandler);
}

private function srvcGetGlobalResourceMappingResultHandler(e:ResultEvent):void {
    if(!chkError(e)) {
        var response:Object = e.result.response;
        var tbl:Object = response.table;
        if(tbl.tablename != tablename || tbl.table_type != table_type) {
            Alert.show("Invalid information", "Error");
        } else {
            var obj:Object;
            resourceMapping = new Array();
            if(tbl.mapping == null) {
                dgGlobalResourceMapping.visible = false;
                dgGlobalResourceMapping.includeInLayout = false;
                lblNoGRMapping.visible = true;
                lblNoGRMapping.includeInLayout = true;
            } else if(tbl.mapping is ObjectProxy) {
                dgGlobalResourceMapping.visible = true;
                dgGlobalResourceMapping.includeInLayout = true;
                obj = new Object();
                obj.id = tbl.mapping.id;
                obj.table_column = tbl.mapping.table_column; 
                obj.resource_tablename = tbl.mapping.resource_tablename; 
                obj.resource_column = tbl.mapping.resource_column;
                resourceMapping.push(obj);
                dgGlobalResourceMapping.rowCount = 1;
            } else {
                dgGlobalResourceMapping.visible = true;
                dgGlobalResourceMapping.includeInLayout = true;
                for each(var mapping:Object in tbl.mapping) {
                    obj = new Object();
                    obj.id = mapping.id;
                    obj.table_column = mapping.table_column; 
                    obj.resource_tablename = mapping.resource_tablename; 
                    obj.resource_column = mapping.resource_column;
                    resourceMapping.push(obj);
                }
                dgGlobalResourceMapping.rowCount = resourceMapping.length;
            }
        }
    }
    UnblockUI();
}

private function srvcHTTPFaultHandler(e:FaultEvent):void
{
    var faultstring:String = e.fault.faultString;
    UnblockUI();
    Alert.show(faultstring, "Error");
}

public function addResourceMapping():void {
    var popup:FormAddEditGlobalResourceMapping = PopUpManager.createPopUp(this, FormAddEditGlobalResourceMapping, true) as FormAddEditGlobalResourceMapping;
    popup.tablename = tablename;
    popup.table_type = table_type;
    popup.callbackFunc = getGlobalResourceMapping;
    PopUpManager.centerPopUp(popup);
}

public function editResourceMapping(mapping:Object):void {
    var popup:FormAddEditGlobalResourceMapping = PopUpManager.createPopUp(this, FormAddEditGlobalResourceMapping, true) as FormAddEditGlobalResourceMapping;
    popup.tablename = tablename;
    popup.table_type = table_type;
    popup.mapping = mapping;
    popup.callbackFunc = getGlobalResourceMapping;
    PopUpManager.centerPopUp(popup); 
}

import mx.controls.Alert;
import mx.events.CloseEvent;
import mx.managers.PopUpManager;
import mx.rpc.events.ResultEvent;

[Bindable]
public var mapping:Object;

[Bindable]
public var mapping_id:int;

[Bindable]
private var resourceTables:Array;

[Bindable]
private var resourceTableColumns:Array;

[Bindable]
public var tablename:String;

[Bindable]
public var table_type:String; 

[Bindable]
private var tableColumns:Array;

[Bindable]
public var callbackFunc:Function;

private var recordSaved:Boolean = false;

private function titleWindow_close(evt:CloseEvent):void {
    closeWindow();
}

private function closeWindow():void {
    if(recordSaved && callbackFunc != null) {
        callbackFunc();
    }
    PopUpManager.removePopUp(this);
}

public function init():void {
    if(mapping != null) {
        mapping_id = mapping.id;
        this.title = "Update mapping"; 
    } else {
        this.title = "Add mapping"; 
    }
    getAllGlobalResources();
    getGRMappedTableColumns();
}

private function getAllGlobalResources():void {
    BlockUI();
    var url:String = srvr_url + ml_data_file;
    var method:String = "POST";
    var xmlParams:XML = 
        <request>
            <action>getAllGlobalResources</action>
            <params/>
        </request>;
    var params:Object = new Object();
    params.request = xmlParams;
    useHttpService(url, method, params, null, srvcGetAllGlobalResourcesResultHandler, srvcHTTPFaultHandler);
}

private function srvcGetAllGlobalResourcesResultHandler(e:ResultEvent):void {
    if(!chkError(e)) {
        var response:Object = e.result.response;
        var rtables:Object = response.resource_tables;
        resourceTables = new Array();
        resourceTableColumns = new Array();
        for each(var resource:Object in rtables.resource) {
            resourceTables.push({'label':resource.tablename, 'data':resource.tablename});
            var arr_cols:Array = getColumnsArray(resource.columns);
            resourceTableColumns[resource.tablename] = arr_cols;
        }
        if(resourceTables.length > 0) {
            cmbResourceTable.selectedIndex = 0;
            cmbResourceColumn.dataProvider = resourceTableColumns[cmbResourceTable.selectedItem.data];
            if(mapping != null) {
                setCBSelectedItem(cmbResourceTable, mapping.resource_tablename);
                setCBSelectedItem(cmbResourceColumn, mapping.resource_column);
            }
        }
    }
    UnblockUI();
}

public function getGRMappedTableColumns():void {
    BlockUI();
    var url:String = srvr_url + ml_data_file;
    var method:String = "POST";
    var xmlParams:XML = 
        <request>
            <action>getGRMappedTableColumns</action>
            <params>
                <table type={table_type}>{tablename}</table>
            </params>
        </request>;
    var params:Object = new Object();
    params.request = xmlParams;
    useHttpService(url, method, params, null, srvcGetGRMappedTableColumnsResultHandler, srvcHTTPFaultHandler);
}

private function srvcGetGRMappedTableColumnsResultHandler(e:ResultEvent):void {
    if(!chkError(e)) {
        var response:Object = e.result.response;
        var tbl:Object = response.table;
        if(tbl.tablename != tablename || tbl.table_type != table_type) {
            Alert.show("Invalid information", "Error");
        } else {
            tableColumns = getColumnsArray(tbl.columns);
            removeMappedTableColumns(tbl.mapped_columns);
            if(mapping != null) {
                tableColumns = tableColumns.concat(getColumnsArray(mapping.table_column));
                setCBSelectedItem(cmbTableColumn, mapping.table_column);
            }
        }
    }
}

public function saveGlobalResourceMapping():void {
    BlockUI();
    var url:String = srvr_url + ml_data_file;
    var method:String = "POST";
    var rtablename:String = cmbResourceTable.selectedItem.data;
    var rcolumn:String = "'" + cmbResourceColumn.selectedItem.data + "'";
    var tcolumn:String = "'" + cmbTableColumn.selectedItem.data + "'";
    var xmlParams:XML = 
        <request>
            <action>saveGlobalResourceMapping</action>
            <params>
                <table tablename={tablename} type={table_type}>
                    <mapping id={mapping_id} resource_tablename={rtablename} resource_column={rcolumn} table_column={tcolumn}/>
                </table>
            </params>
        </request>;
    var params:Object = new Object();
    params.request = xmlParams;
    useHttpService(url, method, params, null, srvcSaveGlobalResourceMappingResultHandler, srvcHTTPFaultHandler);
}

private function srvcSaveGlobalResourceMappingResultHandler(e:ResultEvent):void {
    if(!chkError(e)) {
        recordSaved = true;
        Alert.show(e.result.response.error.msg, "Response");
    } else {
        recordSaved = false;
    }
}

private function srvcHTTPFaultHandler(e:FaultEvent):void
{
    var faultstring:String = e.fault.faultString;
    UnblockUI();
    recordSaved = false;
    Alert.show(faultstring, "Error");
}

private function removeMappedTableColumns(mappedCols:String):void {
    var arr_cols:Array = getColumnsArray(removeSingleQuotes(mappedCols));
    var tlen:int = tableColumns.length;
    var mlen:int = arr_cols.length;
    var remvd:int = 0;
    for(var i:int = 0; i < tlen; i++) {
        for(var j:int = 0; j < mlen; j++) {
            if(arr_cols[j].data == tableColumns[i].data) {
                tableColumns.splice(i, 1);
                remvd++;
                break;
            }
        }
        if(remvd >= mlen) {
            break;
        }
    }
}

private function loadResourceColumns():void {
    cmbResourceColumn.dataProvider = resourceTableColumns[cmbResourceTable.selectedItem.data];
}

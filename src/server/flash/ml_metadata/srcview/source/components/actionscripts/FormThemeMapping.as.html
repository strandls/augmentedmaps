// ActionScript file
import mx.collections.ArrayCollection;
import mx.controls.Alert;
import mx.rpc.events.ResultEvent;
import mx.utils.ObjectProxy;

[Bindable]
public var dataFetched:Boolean = false;

private var mappingIds:Array = new Array();

public function init():void
{
    getAllThemes();
}

private function getAllThemes():void
{
    BlockUI();
    var url:String = srvr_url + ml_data_file;
    var method:String = "POST";
    var xmlParams:XML = 
        <request>
            <action>getAllThemes</action>
            <params/>
        </request>;
    var params:Object = new Object();
    params.request = xmlParams;
    useHttpService(url, method, params, null, srvcGetAllThemesResultHandler, srvcHTTPFaultHandler);
}

private function getThemeMapping():void
{
    BlockUI();
    var url:String = srvr_url + ml_data_file;// + "?action=getThemeMapping&layer_tablename=" + layer_tablename;
    var method:String = "POST";
    var xmlParams:XML = 
        <request>
            <action>getThemeMapping</action>
            <params>
                <layer_tablename>{layer_tablename}</layer_tablename>
            </params>
        </request>;
    var params:Object = new Object();
    params.request = xmlParams;
    useHttpService(url, method, params, null, srvcGetThemeMappingResultHandler, srvcHTTPFaultHandler);
}

private function srvcGetThemeMappingResultHandler(e:ResultEvent):void
{
    if(!chkError(e)) {
        var response:Object = e.result.response;
        var layer:Object = response.layer;
        if (layer.tablename != layer_tablename) {
            Alert.show("Invalid information" + layer.layer_tablename, "Error");
        } else {
            for each(var mapping:Object in layer.mapping)
            {
                if(mapping.theme_type == 1)
                {
                    setCBSelectedItem(cmbBxByTheme, mapping.theme_id);
                    mappingIds[1] = mapping.mapping_id;
                }                    
                else if(mapping.theme_type == 2)
                {
                    setCBSelectedItem(cmbBxByGeography, mapping.theme_id);
                    mappingIds[2] = mapping.mapping_id;
                }
            }
        }
    }
    dataFetched = true;
    UnblockUI();
}

private function srvcGetAllThemesResultHandler(e:ResultEvent):void
{
    if(!chkError(e)) {
        var response:Object = e.result.response;
        for each(var themes:Object in response.themes)
        {
            if(themes.theme_type == 1)
            {
                if(themes.theme != null) {
                    cmbBxByTheme.dataProvider = ThemesObjectToArray(themes.theme);
                }
            }
            else if(themes.theme_type == 2)
            {
                if(themes.theme != null) {
                    cmbBxByGeography.dataProvider = ThemesObjectToArray(themes.theme);
                }
            }
        }
        getThemeMapping();
    }
    UnblockUI();
}

private function ThemesObjectToArray(themes:Object):Array
{
    var arrThemes:Array = new Array();
    var objTheme:Object;
    if(themes == null) {
    } else if(themes is ObjectProxy) {
        objTheme = new Object();
        objTheme.label = themes.theme_name;
        objTheme.data = themes.theme_id;
        arrThemes.push(objTheme);
    } else if(themes is ArrayCollection) {
        for each(var theme:Object in themes) {
            objTheme = new Object();
            objTheme.label = theme.theme_name;
            objTheme.data = theme.theme_id;
            arrThemes.push(objTheme);
        }
    }
    return arrThemes;
}

private function saveThemeMapping():void {
    BlockUI();
    var url:String = srvr_url + ml_data_file;// + "?action=getThemeMapping&layer_tablename=" + layer_tablename;
    var method:String = "POST";
    var xmlParams:XML = 
        <request>
            <action>saveThemeMapping</action>
            <params>
                <layer tablename={layer_tablename}>
                    <theme_mapping id={mappingIds[1]}>{cmbBxByTheme.selectedItem.data}</theme_mapping>
                    <theme_mapping id={mappingIds[2]}>{cmbBxByGeography.selectedItem.data}</theme_mapping>
                </layer>
            </params>
        </request>;
    var params:Object = new Object();
    params.request = xmlParams;
    useHttpService(url, method, params, null, srvcSaveThemeMappingResultHandler, srvcHTTPFaultHandler);
}

private function srvcSaveThemeMappingResultHandler(e:ResultEvent):void {
    if(e.result.b) {
        Alert.show("Server error.", "Error");
        return;
    }
    var response:Object = e.result.response;
    if(response.error.value == 1) {
        Alert.show(response.error.msg, "Error");
    } else {
        Alert.show(response.error.msg, "Response");
    }
}

private function srvcHTTPFaultHandler(e:FaultEvent):void
{
    var faultstring:String = e.fault.faultString;
    UnblockUI();
    Alert.show(faultstring, "Error");
}

import mx.events.IndexChangedEvent;

public function init():void
{
    cmpFormMetaLink.init();
}

private function tabChanged(e:IndexChangedEvent):void {
    switch(tabNavi.selectedChild.id) {
        case 'tabFormMetaLink':
            if(!cmpFormMetaLink.dataFetched)
                cmpFormMetaLink.init();
        break;
        case 'tabFormLinkColumnDesc':
            if(!cmpFormLinkColumnDesc.dataFetched)
                cmpFormLinkColumnDesc.init();
        break;
        case 'tabFormGlobalResourceMapping':
            if(!cmpFormGlobalResourceMapping.dataFetched)
                cmpFormGlobalResourceMapping.init();
        break;
        default:
        break;
    }
}

public function useHttpService(url:String, method:String, parameters:Object, resultFormat:String, resultHandler:Function, faultHandler:Function):void {
    service = new HTTPService();
    service.useProxy = false;
    if (resultFormat != null) {
        service.resultFormat = resultFormat;
       }
    service.url = url;
    service.method = method;
    service.addEventListener(ResultEvent.RESULT, resultHandler);
    service.addEventListener(FaultEvent.FAULT, faultHandler);
    service.send(parameters);
}

private function getFormItem_Text(lbl:String, val:String):FormItem
{
    var frmitm:FormItem = new FormItem();
    frmitm.label = lbl;
    
    var txtBx:TextInput = new TextInput();
    txtBx.id = 'edit-'+lbl;
    txtBx.name = 'edit-'+lbl;
    txtBx.text = val;
    
    frmitm.addChild(txtBx);
    
    return frmitm;
} 

private function setCBSelectedItem(cmbBx:ComboBox, _selectedValue:String):void {
    if(cmbBx.dataProvider != null) {
        // Loop through dataProvider 
        for (var i:int=0; i<cmbBx.dataProvider.length; i++) {
            // Get this item's data 
            var item:String = cmbBx.dataProvider[i].data;
             
            // Check if is selectedValue 
            if(item == _selectedValue) {
                // Yes, set selectedIndex 
                cmbBx.selectedIndex = i;
                break;
            }
        }
    }
}

private function setListSelectedItems(lst:List, vals:String):void {
    if(lst.dataProvider != null) {
        var arr_vals:Array = vals.split(",");
        var lstSelectedIndices:Array = new Array();
        // Loop through dataProvider 
        for (var i:int=0; i<lst.dataProvider.length; i++) {
            // Get this item's data 
            var item:String = lst.dataProvider[i].data;

            if(arr_vals.indexOf(item) != -1) {
                lstSelectedIndices.push(i);
            }
        }
        lst.selectedIndices = lstSelectedIndices;
    }
}

private function initSpecifiedColumns(lst:List, txtBx:TextInput, val:String):void
{
    if (val == null)
    {
        val = '';
    }
    txtBx.text = val;
    setListSelectedItems(lst, removeSingleQuotes(val));
}

private function updateSelectedColumnsList(lst:List, txtIp:TextInput):void {
    // For ref: current clicked item can be passed using "List(event.target).selectedItem"
    var str:String = '';
    for each(var obj:Object in lst.selectedItems) {
        if(obj.data == null || obj.data == '')
        {
            initSpecifiedColumns(lst, txtIp, null);
            return;
        }

        str += "'" + obj.data + "',";
    }
    if(str.length > 0) {
        str = str.substr(0, str.length-1);
    }
    txtIp.text = str;
}

private function isFilterableChanged(frmItmFilterColumns:FormItem, chkBx:CheckBox):void
{
    if (chkBx.selected)
    {
        frmItmFilterColumns.visible = true;
        frmItmFilterColumns.includeInLayout = true;
    }
    else
    {
        frmItmFilterColumns.visible = false;
        frmItmFilterColumns.includeInLayout = false;
    }
}

private function getColumnsArray(str_cols:String):Array
{
    str_cols = removeSingleQuotes(str_cols);
    var table_columns:Array = [];
    var col:String;
    var objCol:Object;
    var arr_columns:Array = str_cols.split(",");
    for each(col in arr_columns) {
        if (col.substr(0, auto_columns_prefix.length) != auto_columns_prefix) {
            objCol = new Object();
            objCol.label = col;
            objCol.data = col;
            table_columns.push(objCol);
        }
    }
    return table_columns;
}

public function chkError(e:ResultEvent):Boolean {
    if(e.result.b) {
        Alert.show("Server error.", "Error");
        return true;
    }
    var response:Object = e.result.response;
    if(response.error.value == 1) {
        Alert.show(response.error.msg, "Error");
        return true;
    }
    return false;
}

private function removeSingleQuotes(str:String):String
{
    if (str == null || str == '')
    {
        return '';
    }
    else
    {
        return str.replace(/'/g, ""); 
    }
}

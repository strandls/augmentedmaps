    {label:"LINESTRING", data:"LINESTRING"},
    {label:"MULTILINESTRING", data:"MULTILINESTRING"},
    {label:"RASTER", data:"RASTER"},
];

public var layer_status: Array = [
    {label:"Inactive", data:0},
    {label:"Active", data:1},
];

[Bindable]
public var layer_access: Array = [
    {label:"Non-downloadable", data:0},
    {label:"Downloadable", data:1},
];

public var layer_license: Array = [
    {label:"None", data:""},
    {label:"(by)", data:"(by)"},
    {label:"(by-sa)", data:"(by-sa)"},
    {label:"(by-nd)", data:"(by-nd)"},

    {label:"(c)", data:"(c)"},

public var layer_participation_type: Array = [
    {label:"Non-participative", data:0},

];

[Bindable]
public var layer_columns: Array = [];

[Bindable]
public var dataFetched:Boolean = false;

public function init():void {
    getMetaLayer();
}

private function getMetaLayer():void {
    BlockUI();
    var url:String = srvr_url + ml_data_file;// + "?action=getMetaLayer&layer_tablename=" + layer_tablename;
    var method:String = "POST";
    var xmlParams:XML = 
        <request>
            <action>getMetaLayer</action>
            <params>
                <layer_tablename>{layer_tablename}</layer_tablename>
            </params>
        </request>;
    var params:Object = new Object();
    params.request = xmlParams;
    useHttpService(url, method, params, null, srvcGetMetaLayerResultHandler, srvcMetaLayerFaultHandler);
}

private function saveMetaLayer():void {
    var xml_params:XML = 
    <request>
        <action>saveMetaLayer</action>
        <params>
            <layer tablename={layer_tablename}>
                <layer_name>{txtLayerName.text}</layer_name>
                <layer_description>{txtDescription.text}</layer_description>
                <status>{cmbStatus.selectedItem.data}</status>
                <min_scale>{txtMinScale.text}</min_scale>
                <max_scale>{txtMaxScale.text}</max_scale>
                <pdf_link>{txtPDFLink.text}</pdf_link>
                <url>{txtURLLink.text}</url>
                <aggregation>{txtAggregation.text}</aggregation>
                <attribution>{txtAttribution.text}</attribution>
                <lineage>{txtLineage.text}</lineage>
                <tags>{txtTags.text}</tags>
                <license>{cmbLicense.selectedItem.data}</license>
                <comments>{txtComments.text}</comments>
                <summary_columns>{txtSummaryColumns.text}</summary_columns>
                <layer_type>{cmbLayerType.selectedItem.data}</layer_type>
                <access>{cmbAccess.selectedItem.data}</access>
                <is_filterable>{Number(chkIsFilterable.selected)}</is_filterable>
                <filter_columns>{txtFilterColumns.text}</filter_columns>
                <search_columns>{txtSearchColumns.text}</search_columns>
                <color_by>{txtColorByColumn.text}</color_by>
                <title_column>{txtTitleColumn.text}</title_column>
                <size_by>{txtSizeByColumn.text}</size_by>
                <editable_columns>{txtEditableColumns.text}</editable_columns>
                <media_columns>{txtMediaColumns.text}</media_columns>
                <video_columns>{txtVideoColumns.text}</video_columns>
                <italics_columns>{txtItalicsColumns.text}</italics_columns>
                <participation_type>{cmbParticipationType.selectedItem.data}</participation_type>
                <created_by>{txtCreatedBy.text}</created_by>
                <created_date>{txtCreatedDate.text}</created_date>
                <modified_by>{txtModifiedBy.text}</modified_by>
                <modified_date>{txtModifiedDate.text}</modified_date>
                <colorby_cat_col>{txtColorByCatColumns.text}</colorby_cat_col>
            </layer>
        </params>
    </request>;

    var params:Object = new Object();
    params.request = xml_params;
    
    var url:String = srvr_url + ml_data_file;// + "?action=saveMetaLayer&layer_tablename=" + layer_tablename;
    var method:String = "POST";
    useHttpService(url, method, params, null, srvcSaveMetaLayerResultHandler, srvcMetaLayerFaultHandler);
}

private function resetMetaLayer():void {
    getMetaLayer();
}
private function SetDefaultColors():void{
     var instance:DefaultColor = PopUpManager.createPopUp(this, DefaultColor, true) as DefaultColor;                                                              
     PopUpManager.centerPopUp(instance);
     instance.ShowColors(srvr_url,layer_tablename,txtColorByCatColumns.text);                                                                          
               
}
private function srvcGetMetaLayerResultHandler(e:ResultEvent):void {
    if(!chkError(e)) {
        var response:Object = e.result.response;
        var layer:Object = response.layer;
        if (layer.tablename != layer_tablename) {
            Alert.show("Invalid information", "Error");
        } else {
            layer_columns = [
                {label: 'None', data: ''}
            ];
            layer_columns = layer_columns.concat(getColumnsArray(layer.columns));

            txtLayerName.text = layer.layer_name.value;
            txtDescription.text = layer.layer_description.value;
            setCBSelectedItem(cmbLayerType, layer.layer_type.value);
            setCBSelectedItem(cmbStatus, layer.status.value);
            setCBSelectedItem(cmbParticipationType, layer.participation_type.value);
            setCBSelectedItem(cmbAccess, layer.access.value);
            txtMinScale.text = layer.min_scale.value;
            txtMaxScale.text = layer.max_scale.value;
            txtPDFLink.text = layer.pdf_link.value;
            txtURLLink.text = layer.url.value;
            txtAggregation.text = layer.aggregation.value;
            txtAttribution.text = layer.attribution.value;
            txtLineage.text = layer.lineage.value;
            txtTags.text = layer.tags.value;
            setCBSelectedItem(cmbLicense, layer.license.value);
            txtComments.text = layer.comments.value;
            txtCreatedBy.text = layer.created_by.value;
            txtCreatedDate.text = layer.created_date.value;
            txtModifiedBy.text = layer.modified_by.value;
            txtModifiedDate.text = layer.modified_date.value;
            initSpecifiedColumns(lstSummaryColumns, txtSummaryColumns, layer.summary_columns.value);
            if (layer.is_filterable.value == 1) {
                chkIsFilterable.selected = true;
                frmItmFilterColumns.visible = true;
                frmItmFilterColumns.includeInLayout = true;
            } else {
                chkIsFilterable.selected = false;
                frmItmFilterColumns.visible = false;
                frmItmFilterColumns.includeInLayout = false;
            }

            initSpecifiedColumns(lstFilterColumns, txtFilterColumns, layer.filter_columns.value);

            initSpecifiedColumns(lstSearchColumns, txtSearchColumns, layer.summary_columns.value);

            initSpecifiedColumns(lstColorByColumn, txtColorByColumn, layer.color_by.value);

            initSpecifiedColumns(lstSizeByColumn, txtSizeByColumn, layer.size_by.value);

            initSpecifiedColumns(lstEditableColumns, txtEditableColumns, layer.editable_columns.value);

            initSpecifiedColumns(lstTitleColumn, txtTitleColumn, layer.title_column.value);

            initSpecifiedColumns(lstMediaColumns, txtMediaColumns, layer.media_columns.value);

            initSpecifiedColumns(lstVideoColumns, txtVideoColumns, layer.video_columns.value);

            initSpecifiedColumns(lstItalicsColumns, txtItalicsColumns, layer.italics_columns.value);

            initSpecifiedColumns(lstColorByCatColumns, txtColorByCatColumns, layer.colorby_cat_col.value);
            
            btnSaveMetaLayer.enabled = true;
        }
    }
    dataFetched = true;
    UnblockUI();
}

private function srvcSaveMetaLayerResultHandler(e:ResultEvent):void {
    if(!chkError(e)) {
        Alert.show(e.result.response.error.msg, "Response");
    }
}

private function srvcMetaLayerFaultHandler(e:FaultEvent):void {
    var faultstring:String = e.fault.faultString;
    UnblockUI();
    Alert.show(faultstring, "Error");
}

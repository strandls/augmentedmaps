// ActionScript file
import mx.binding.utils.BindingUtils;
import mx.containers.FormItem;
import mx.controls.Alert;
import mx.rpc.events.FaultEvent;
import mx.rpc.events.ResultEvent;

[Bindable]
public var dataFetched:Boolean = false;

[Bindable]
public var cols_desc:Array;

public function init():void
{
    getLinkColDesc();
}

private function getLinkColDesc():void
{
    BlockUI();
    FrmLinkColDesc.removeAllChildren();
    var url:String = srvr_url + ml_data_file;// + "?action=getMetaLayer&layer_tablename=" + layer_tablename;
    var method:String = "POST";
    var xmlParams:XML = 
        <request>
            <action>getLinkColDesc</action>
            <params>
                <link_tablename>{link_tablename}</link_tablename>
            </params>
        </request>;
    var params:Object = new Object();
    params.request = xmlParams;
    useHttpService(url, method, params, null, srvcGetLinkColDescResultHandler, srvcLinkColDescFaultHandler);
}

private function srvcGetLinkColDescResultHandler(e:ResultEvent):void
{
    if(!chkError(e)) {
        var response:Object = e.result.response;
        cols_desc = new Array();
        var link:Object = response.link;
        if (link.tablename != link_tablename) {
            Alert.show("Invalid information" + link.tablename, "Error");
        } else {
            for each(var col:Object in link.column)
            {
                cols_desc.push(col);
                var frmItm:FormItem = getFormItem_Text(col.c_name, col.c_desc);
                FrmLinkColDesc.addChild(frmItm);
                // Binding a UI element to a variable. Whenever the value in UI element is changed, 
                // the value is updated in the variable. 
                // BindingUtils.bindProperty(destination, 'property', UIelem, 'uiproperty(usually text)');
                BindingUtils.bindProperty(cols_desc[cols_desc.length - 1], 'c_desc', frmItm.getChildAt(0), 'text');
            }
        }
    }
    dataFetched = true;
    btnSubmit.enabled = true;
    UnblockUI();
}

private function saveLinkColDesc():void {
    BlockUI();
    var url:String = srvr_url + ml_data_file;
    var method:String = "POST";
    var xmlParams:XML = 
        <request>
            <action>saveLinkColDesc</action>
            <params>
                <link tablename={link_tablename}/>
            </params>
        </request>;
    var lnkXML:XML = xmlParams.child("params")[0].child("link")[0];

    var len:int = cols_desc.length;
    for(var i:int = 0; i < len; i++) {
        // Since a binding has been done on the text boxes to the cols_desc array, we get the values which are in form.
        lnkXML.appendChild(<{cols_desc[i].c_name}>{cols_desc[i].c_desc}</{cols_desc[i].c_name}>);
    }
    var params:Object = new Object();
    params.request = xmlParams;
    useHttpService(url, method, params, null, srvcSaveLinkColDescResultHandler, srvcLinkColDescFaultHandler);
}

private function srvcSaveLinkColDescResultHandler(e:ResultEvent):void {
    if(!chkError(e)) {
        Alert.show(e.result.response.error.msg, "Response");
    }
}

private function srvcLinkColDescFaultHandler(e:FaultEvent):void
{
    var faultstring:String = e.fault.faultString;
    UnblockUI();
    Alert.show(faultstring, "Error");
}

// ActionScript file
import mx.controls.Alert;
import mx.rpc.events.FaultEvent;
import mx.rpc.events.ResultEvent;

[Bindable]
private var NONE_ARRAY:Array = [
    {label: 'None', data: ''}
];

[Bindable]
private var link_columns:Array = [];

[Bindable]
private var layer_columns:Array = [];

[Bindable]
public var link_status: Array = [
    {label:"Inactive", data:0},
    {label:"Active", data:1},
];

[Bindable]
public var dataFetched:Boolean = false;

public function init():void
{
    getMetaLink();
}

private function getMetaLink():void
{
    BlockUI();
    var url:String = srvr_url + ml_data_file;// + "?action=getMetaLinkTable&link_tablename=" + link_tablename;
    var method:String = "POST";
    var xmlParams:XML = 
        <request>
            <action>getMetaLinkTable</action>
            <params>
                <link_tablename>{link_tablename}</link_tablename>
            </params>
        </request>;
    var params:Object = new Object();
    params.request = xmlParams;
    useHttpService(url, method, params, null, srvcGetMetaLinkTableResultHandler, srvcMetaLinkTableFaultHandler);
}

private function srvcGetMetaLinkTableResultHandler(e:ResultEvent):void
{
    if(!chkError(e)) {
        var response:Object = e.result.response;
        var layer:Object = response.layer;
        var link:Object = response.link;
        if (layer.tablename != layer_tablename || link.tablename != link_tablename ) {
            Alert.show("Invalid information", "Error");
        } else {
            var table_columns:Array;
            var col:String;
            var objCol:Object;
            layer_columns = getColumnsArray(layer.columns);
            
            link_columns = getColumnsArray(link.columns);
            
            txtLinkID.text = getNodeValue(link.id);
            txtLinkName.text = getNodeValue(link.link_name);
            txtDescription.text = getNodeValue(link.description);
            setCBSelectedItem(cmbStatus, getNodeValue(link.status));
            txtCreatedBy.text = getNodeValue(link.created_by);
            txtCreatedDate.text = getNodeValue(link.created_date);
            txtModifiedBy.text = getNodeValue(link.modified_by);
            txtModifiedDate.text = getNodeValue(link.modified_date);
            initSpecifiedColumns(lstSummaryColumns, txtSummaryColumns, getNodeValue(link.summary_columns));
            initSpecifiedColumns(lstSearchColumns, txtSearchColumns, getNodeValue(link.search_columns));
            initSpecifiedColumns(lstLinkedColumn, txtLinkedColumn, getNodeValue(link.linked_column));
            initSpecifiedColumns(lstLayerColumn, txtLayerColumn, getNodeValue(link.layer_column));
            initSpecifiedColumns(lstEditableColumns, txtEditableColumns, getNodeValue(link.editable_columns));
            initSpecifiedColumns(lstItalicsColumns, txtItalicsColumns, getNodeValue(link.italics_columns));
            initSpecifiedColumns(lstMandatoryColumns, txtMandatoryColumns, getNodeValue(link.mandatory_columns));
            initSpecifiedColumns(lstRecordTypeColumn, txtRecordTypeColumn, getNodeValue(link.record_type_column));
            if (link.is_filterable.value == 1) {
                chkIsFilterable.selected = true;
                frmItmFilterColumns.visible = true;
                frmItmFilterColumns.includeInLayout = true;
            } else {
                chkIsFilterable.selected = false;
                frmItmFilterColumns.visible = false;
                frmItmFilterColumns.includeInLayout = false;
            }
            initSpecifiedColumns(lstFilterColumns, txtFilterColumns, getNodeValue(link.filter_columns));
        }
    }
    dataFetched = true;
    UnblockUI();
}

private function saveMetaLinkTable():void {
    var xmlParams:XML = 
        <request>
            <action>saveMetaLinkTable</action>
            <params>
                <link tablename={link_tablename}>
                    <description>{txtDescription.text}</description>
                    <status>{cmbStatus.selectedItem.data}</status>
                    <summary_columns>{txtSummaryColumns.text}</summary_columns>
                    <linked_column>{txtLinkedColumn.text}</linked_column>
                    <layer_column>{txtLayerColumn.text}</layer_column>
                    <search_columns>{txtSearchColumns.text}</search_columns>
                    <editable_columns>{txtEditableColumns.text}</editable_columns>
                    <is_filterable>{Number(chkIsFilterable.selected)}</is_filterable>
                    <filter_columns>{txtFilterColumns.text}</filter_columns>
                    <link_name>{txtLinkName.text}</link_name>
                    <italics_columns>{txtItalicsColumns.text}</italics_columns>
                    <mandatory_columns>{txtMandatoryColumns.text}</mandatory_columns>
                    <record_type_column>{txtRecordTypeColumn.text}</record_type_column>
                    <created_by>{txtCreatedBy.text}</created_by>
                    <created_date>{txtCreatedDate.text}</created_date>
                    <modified_by>{txtModifiedBy.text}</modified_by>
                    <modified_date>{txtModifiedDate.text}</modified_date>
                </link>
            </params>
        </request>;

    var params:Object = new Object();
    params.request = xmlParams;
    
    var url:String = srvr_url + ml_data_file;
    var method:String = "POST";
    useHttpService(url, method, params, null, srvcSaveMetaLayerResultHandler, srvcMetaLinkTableFaultHandler);
}

private function srvcSaveMetaLayerResultHandler(e:ResultEvent):void {
    if(!chkError(e)) {
        Alert.show(e.result.response.error.msg, "Response");
    }
}

private function srvcMetaLinkTableFaultHandler(e:FaultEvent):void
{
    var faultstring:String = e.fault.faultString;
    UnblockUI();
    Alert.show(faultstring, "Error");
}
